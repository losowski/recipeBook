\begin{method}
\PreparationTitle{Par-boil \NewPotatoes}
\PreparationStep{Warm fan-assisted oven to \Temperature{180} }
\IngredientPreparation{\NewPotatoes}{Cut into 1-inch chunks}
\PreparationStep{Place \NewPotatoes in a pot for par-boiling}
\PreparationStep{Add cold water, cover}
\CookingStep{Boil at high heat}
\CookingCondition{12-minutes elapsed from water boiled}

\PreparationTitle{Brown \Onion}
\IngredientPreparation{\Onion}{Finely cut}
\IngredientPreparation{\SmokedBacon}{Cut into 1cm squares}
\CookingStep{Fry at medium heat}
\CookingCondition{\Onion turn translucent}

\PreparationTitle{Prepare for skillet}
\IngredientPreparation{\BellPepper}{Cut into chunks}
\IngredientPreparation{\ChestnutMushrooms}{Cut into quarters}
\IngredientPreparation{\Chorizo}{Peel and cut into 1cm quarters}

\IngredientPreparation{\OliveOil}{sprinkle}
\IngredientPreparation{\Cumin}{\add}
\IngredientPreparation{\SmokedPakrika}{\add}
\IngredientPreparation{\ChipotlePaste}{\add}
\IngredientPreparation{\Salt}{\add}
\IngredientPreparation{\GroundBlackPepper}{sprinkle}
\IngredientPreparation{\Garlic}{Crush}
\PreparationStep{\ChestnutMushrooms}{\add}
\PreparationStep{\Chorizo}{\add}
\PreparationStep{\BellPepper}{\add}
\CookingStep{Fry at medium heat}
