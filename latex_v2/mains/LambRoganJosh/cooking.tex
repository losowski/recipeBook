% Cooking Instructions
\begin{method}

% Zeroth Round
\PreparationTitle{Prepare the vegetables}
\IngredientPreparation{\Onion}{Peel, and cut into quarters}
\IngredientPreparation{\Garlic}{Peel and cut finely}
\IngredientPreparation{\Ginger}{Peel and cut finely}



% First round
\PreparationTitle{Brown Onions}
\CookingStep{Using \Skillet / \FryingPan}
\IngredientPreparation{\OliveOil}{sprinkle}
\IngredientPreparation{\Onion}{\add}
\IngredientPreparation{\Garlic}{\add}
\IngredientPreparation{\Ginger}{\add}
\IngredientPreparation{\Almonds}{\add}
\IngredientPreparation{\Raisins}{\add}
\CookingStep{Fry at medium heat}
\CookingCondition{\Onion turn translucent}
\CookingStep{Transfer to \LargePot}


% Second round
\PreparationTitle{Brown the \DicedLamb}
\CookingStep{Using \Skillet / \FryingPan}
\IngredientPreparation{\DicedLamb}{\add}
\IngredientPreparation{\OliveOil}{sprinkle}
\CookingStep{Fry at high heat}
\CookingCondition{\DicedLamb is browned outside}
\CookingStep{Transfer to \PressureCooker}


% Third round
\PreparationTitle{Prepare the stock}
\IngredientPreparation{\Paprika}{\add}
\IngredientPreparation{\MadrasCurryPowder}{\add}
\IngredientPreparation{\GroundCinnamon}{\add}
\IngredientPreparation{\Cardamon}{\add}
\IngredientPreparation{\Cloves}{\add}
\IngredientPreparation{\Bayleaves}{\add}
\IngredientPreparation{\Tumeric}{\add}

\IngredientPreparation{\FennelSeeds}{\add}
\IngredientPreparation{\GaramMasala}{\add}
\IngredientPreparation{\TomatoPuree}{\add}
\IngredientPreparation{\NaturalYogurt}{\add}
\IngredientPreparation{\ChickenStock}{\add}


% Fourth round
\PreparationTitle{\PressureCooker \DicedLamb}
\CookingStep{High pressure setting on Pressure Cooker}
\CookingCondition{15 mins has elapsed}


% Fifth round
\PreparationTitle{\LargePot}
\CookingStep{Cook on medium-low heat}
\CookingCondition{30+ mins has elapsed and meat flakes off bones}
