% Vegetable preparation
\PreparationTitle{Vegetable Preparation}

\IngredientPreparation{\Carrot}{chopped roughly}

\IngredientPreparation{\NewPotatoes}{chopped into halves}


% Pressure Cook Vegetables
\PreparationTitle{Pressure Cook Vegtables}
\AddIngredient{\NewPotatoes}
\AddIngredient{\Carrot}

\CookingCondition{9 mins has elapsed}

\PreparationTitle{Cook \Onion}
\AddIngredient{\BabyCupMushrooms}
\AddIngredient{\Onion}
\AddIngredient{Stock from beef cooking}
\CookingStep{Cover and cook in pot at medium-low heat}
\CookingCondition{20-minutes elapsed, and meat is tender}


% Brown meats
\PreparationTitle{Brown \DicedBeef}
\PreparationStep{Using skillet}
\PreparationStep{\DicedBeef}
\AddIngredient{\Garlic}
\AddIngredient{\WorcesterSauce}
\AddIngredient{\Salt}{applied generously to meat}
\AddIngredient{\GroundBlackPepper}{applied generously to meat}
\CookingStep{Fry at high heat}
\CookingCondition{\DicedBeef is golden outside}


\PreparationTitle{Combine Ingredients}
\AddIngredient{\NewPotatoes}
\AddIngredient{\Carrot}
\AddIngredient{\BabyCupMushrooms}
\AddIngredient{\Onion}
