% One

% Two
\PreparationTitle{Preparation of \DicedBeef for saute}
\PreparationStep{Using \LargePot}
\IngredientPreparation{\Flour}{Pour into dry \LargePot}
\IngredientPreparation{\DicedBeef}{Cover each peice of meat in flour}

% Three

\IngredientPreparation{\Carrot}{Sliced into 5mm chunks}
\IngredientPreparation{\Potatoes}{Cut into 1-inch chunks}
\IngredientPreparation{\Celery}{Sliced into 5mm chunks}

% Four
\PreparationTitle{Make Up stock}
\PreparationStep{\Salt}{\add}

\PreparationStep{\SoySauce}{\add}
\PreparationStep{\SrirachaHotSauce}{\add}

% Five
\PreparationTitle{Saute the beef}

\PreparationStep{\DicedBeef}{\add}
\IngredientPreparation{\OliveOil}{sprinkle}

\CookingCondition{60 mins has elapsed}
