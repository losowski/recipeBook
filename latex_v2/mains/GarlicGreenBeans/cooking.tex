% Cooking Instructions
\begin{method}
\PreparationTitle{Par-boil Potatoes}
\IngredientPreparation{\NewPotatoes}{Cut into 1-inch chunks}
\PreparationStep{Place potatoes in a pot for par-boiling}
\PreparationStep{Add cold water, cover}
\CookingStep{Boil at high heat}
\CookingCondition{12-minutes elapsed from water boiled}


\PreparationTitle{Prepare tray Ingredients}
\PreparationStep{Warm fan-assisted oven to \Temperature{180} }
\IngredientPreparation{\GreenBeans}{Remove stalks and cut into 1-inch long peices}
\IngredientPreparation{White onion}{Peel and slice roughly into strips}
\IngredientPreparation{\Garlic}{Crush}
\IngredientPreparation{\Carrot}{Cut into quarter chunks roughly 1cm thick}
\IngredientPreparation{\ChestnutMushrooms}{Cut into quarters}
\IngredientPreparation{\Chorizo}{Peel and cut into 1cm quarters}

% Stock
\PreparationTitle{Prepare Stock}
\IngredientPreparation{Pork Stock}{Dissolve 1 stock cube in 400ml water}


\PreparationTitle{Tray preparation}
\PreparationStep{\ButterBeans}{Add}
\PreparationStep{\GreenBeans}{Add}
\PreparationStep{White onion}{Add}
\PreparationStep{\Garlic}{Add}
\PreparationStep{\Carrot}{Add}
\PreparationStep{\ChestnutMushrooms}{Add}
\PreparationStep{\Chorizo}{Add}
\PreparationStep{Pork Stock}{Add}
\CookingStep{Add boiled potatoes}

\PreparationTitle{Seasoning}
\PreparationStep{\OliveOil}{Spread over the chicken and potatoes}
\PreparationStep{\Salt}{Season x5 waves}
\PreparationStep{\SmokedPakrika}{Disperse}
\PreparationStep{Fennel Seeds}{Disperse}
\PreparationStep{Garlic Granules}{Disperse}
\CookingStep{Mix as thoroughly as possible}

\PreparationTitle{Bake first round}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{20-minutes elapsed}

\PreparationTitle{Final additions preparation}
\IngredientPreparation{Bell Peppers}{Sliced roughly}
\IngredientPreparation{Feta Cheese}{Cut roughly into chunks}
\PreparationStep{Feta Cheese}{Add}
\PreparationStep{Bell Peppers}{Add}

\PreparationTitle{Bake second round}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{15-minutes elapsed}

\end{method}
