% Cooking Instructions
\begin{method}

% First round
\PreparationTitle{Prepare Potatoes}
\IngredientPreparation{\SweetPotatoes}{peel}
\IngredientPreparation{\SweetPotatoes}{Cut into 1-inch chunks}
\PreparationStep{Move to \PressureCooker}
\CookingStep{High pressure setting on Pressure Cooker}
\CookingCondition{15 mins has elapsed}


% Second round
\PreparationTitle{Preparation of ingredients}
\IngredientPreparation{\ChorizoSausage}{Cut finely}
\IngredientPreparation{\Onion}{Cut finely}
\IngredientPreparation{\Garlic}{Crush}
\IngredientPreparation{\Eggs}{Scramble}

% Third round 
\PreparationTitle{Mix}
\AddIngredient{\ChorizoSausage}
\AddIngredient{\Onion}
\AddIngredient{\Garlic}
\AddIngredient{\Eggs}
\AddIngredient{\Flour}
\AddIngredient{\Salt}
\AddIngredient{\Pepper}
\AddIngredient{\SmokedPaprika}

\PreparationTitle{Roll to make croquettes}
\PreparationStep{Roll about 1 table spoon into a sausage shape}
\PreparationStep{\BreadCrumbs}{Roll the sausage shape, and cover completely}
\PreparationStep{\OliveOil}{Spray each croquette}
\PreparationStep{Place into baking tray}


\PreparationTitle{Oven Tray first}
\CookingStep{Cook at \Temperature{180} }
\CookingCondition{25-minutes elapsed until croquettes are browned}


\end{method}
