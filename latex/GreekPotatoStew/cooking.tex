\MethodTitle
% Cooking Instructions
\begin{enumerate}

\PreparationTitle{Initial preparations}
\IngredientPreparation{Onion}{Peeled and sliced}
\IngredientPreparation{Bell Pepper}{De-seeded and sliced}
\IngredientPreparation{Garlic}{Peeled and crushed}
\IngredientPreparation{New Potatoes}{Cut into 2cm Chunks}
\IngredientPreparation{Lemon}{Cut in half, and squeeze lemon}
\IngredientPreparation{Feta Cheese}{Cut into 1cm cubes}

% First round
\PreparationTitle{Saute Peppers}
\PreparationStep{Using skillet/frying pan}
\PreparationStep{Olive Oil}{Spread on hot pan}
\PreparationStep{Onion}{Add}
\PreparationStep{Bell Pepper}{Add}
\CookingStep{Fry at medium heat}
\CookingCondition{Onions turn translucent, and pepper softens}

% Stock
\PreparationTitle{Prepare Vegetable Stock}
\IngredientPreparation{Vegetable Stock}{Dissolve 1 stock cube in 300ml water}

% Second round
\PreparationTitle{Cook garlic}
\PreparationStep{Garlic}{Add}
\PreparationStep{Dried oregano}{Add}
\CookingStep{Fry at medium heat}
\CookingCondition{Garlic flavour is spread nicely}
\PreparationStep{Move to cooking pot}

% Third round
\PreparationTitle{Cook potatoes}
\PreparationStep{Chopped Tomato Tin}{Open and pour in}
\PreparationStep{Vegetable Stock}{Wash out tomato tin, and add}
\PreparationStep{New Potatoes}{Add}
\PreparationStep{Chipotle Sauce}{Add}
\CookingStep{Cover, and simmer at medium low heat}
\CookingCondition{25 minutes elapsed}

% Fourth round
\PreparationTitle{Add olives}
\PreparationStep{Pitted Olives}{Add}
\PreparationStep{Lemon Juice}{Add}
\CookingStep{Uncover, and continue simmering}
\CookingCondition{Potatoes cooked - a knife can slice them}
% Fifth round
\PreparationTitle{Finishing touches}
\PreparationStep{Feta Cheese}{Add as seasoning}

\EndCookingMethod
