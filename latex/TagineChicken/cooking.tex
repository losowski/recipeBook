\begin{enumerate}
\PreparationTitle{Par-boil Potatoes}
\IngredientPreparation{New Potatoes}{Cut into 1-inch chunks}
\PreparationStep{Place potatoes in a pot for par-boiling}
\PreparationStep{Add cold water, cover}
\CookingStep{Boil at high heat}
\CookingCondition{12-minutes elapsed from water boiled}

\PreparationTitle{Prepare other vegetables}
\PreparationStep{Warm fan-assisted oven to \ovenTemp{180} }
\IngredientPreparation{Onion}{Peel, and cut into quarters}
\IngredientPreparation{Garlic}{Peel and leave whole}
\PreparationStep{Dried apricots}{Add}

\PreparationTitle{Baking tray preparation}
\PreparationStep{Prepare large baking tray}
\PreparationStep{Boneless Chicken Thighs}{Spread chicken out across tray}
\PreparationStep{Worchester Sauce}{sprinkle generously over chicken}
\PreparationStep{Fresh Thyme}{Sprinkle}
\CookingStep{Add boiled potatoes}

\PreparationTitle{Bake Chicken}
\PreparationStep{Olive Oil}{Spread over the chicken and potatoes}
\PreparationStep{Salt}{Season}
\PreparationStep{Black pepper}{Sprinkle}
\PreparationStep{Majoram}{Sprinkle}
\PreparationStep{Ras El Hanout}{Sprinkle}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{20-minutes elapsed}

\PreparationTitle{Bake all ingredients}
\CookingStep{Ensure chicken and potatoes rotated}
\PreparationStep{SweetCorn}{Add, including the water from the can}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{20-minutes elapsed}

\IngredientPreparation{Kale}{Wash in a large pot to ensure all leaves are wet}

\PreparationTitle{Add Kale}
\PreparationStep{Juu}{Baste the contents of the tray}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{3-minutes elapsed}

\PreparationTitle{Finishing touches}
\PreparationStep{Juu}{Baste the contents of the tray}
\IngredientPreparation{Lemon}{Freshly squeeze juice}
\PreparationStep{Lemon}{Pour juice over the bake}
