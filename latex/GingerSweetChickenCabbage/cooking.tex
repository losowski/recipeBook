\MethodTitle
% Cooking Instructions
\begin{enumerate}
\PreparationTitle{Par-boil Potatoes}
\IngredientPreparation{New Potatoes}{Cut into 1-inch chunks}
\PreparationStep{Place potatoes in a pot for par-boiling}
\PreparationStep{Add cold water, cover}
\CookingStep{Boil at high heat}
\CookingCondition{15-minutes elapsed from water boiled}


\PreparationTitle{Prepare Vegetables for tray}
\IngredientPreparation{Red Onion}{Cut into strips}
\IngredientPreparation{Garlic}{Cut up}
\IngredientPreparation{Green Beans}{Remove stalks and cut into 1-inch long peices}
\IngredientPreparation{Chestnut Mushrooms}{Cut into quarters}
\IngredientPreparation{Courgette}{Slice into 1cm long half-moons}
\IngredientPreparation{Fresh Ginger}{Cut finely}


\PreparationTitle{Prepare Stock}
\IngredientPreparation{Vegetable Stock}{Dissolve 1 stock cube in 400ml water}


\PreparationTitle{Season the potatoes}
\PreparationStep{Salt}{sprinkle}
\PreparationStep{Olive Oil}{sprinkle}


\PreparationTitle{Season the chicken on the tray}
\IngredientPreparation{Boneless Chicken Thighs}{Spread chicken out across tray}
\PreparationStep{Rice Vinegar}{sprinkle generously over chicken}
\IngredientPreparation{Boneless Chicken Thighs}{Flip the chicken}
\PreparationStep{Rice Vinegar}{sprinkle generously over chicken (again)}
\PreparationStep{Salt}{Season}
\PreparationStep{Black Pepper}{Season}
\PreparationStep{Mustard}{Sprinkle}
\PreparationStep{Ground Fenugreek}{Sprinkle}
\PreparationStep{Ground Cinnamon}{Sprinkle}
\PreparationStep{Caraway Seeds}{Sprinkle}
\PreparationStep{Star Anise}{Sprinkle}


\PreparationTitle{Build up the tray}
\PreparationStep{Red Onion}{Add}
\PreparationStep{Garlic}{Add}
\PreparationStep{Green Beans}{Add}
\PreparationStep{Chestnut Mushrooms}{Add}
\PreparationStep{Courgette}{Add}
\PreparationStep{Fresh Ginger}{Add}


\PreparationTitle{Cabbage Preparation}
\IngredientPreparation{Sweetheart Cabbage}{Cut into 1cm chunks}
\IngredientPreparation{Sweetheart Cabbage}{Rinse in a pot of water}


\PreparationTitle{Cook period 1}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{25-minutes elapsed}


\PreparationTitle{Add the potatoes}
\PreparationStep{Tray contents}{Mix thoroughly}
\PreparationStep{Seasoned potatoes}{Add on top of the tray contents}


\PreparationTitle{Cook period 2}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{25-minutes elapsed}


\PreparationTitle{Cabbage final ingredients}
\PreparationStep{Sweetheart Cabbage}{Add}
\CookingStep{Give ingredients a good stir and basting}


\PreparationTitle{Cook period 2}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{10-minutes elapsed}


\EndCookingMethod
