% Cooking Instructions
\begin{enumerate}
\PreparationTitle{Preparation of meats}
\PreparationStep{Using skillet}
\IngredientPreparation{Diced Pork}{Diced into 1.5cm cubes}
\IngredientPreparation{Onion}{Diced finely}
\IngredientPreparation{Garlic}{Peeled and chopped roughly}
\IngredientPreparation{Smoked Bacon}{Sliced into small pieces (max 1cm squares)}

\PreparationTitle{Brown Onions}
\PreparationStep{Butter}{Melted onto pan}
\PreparationStep{Onion}{Add}
\CookingStep{Fry at medium heat}
\CookingCondition{Onions turn translucent}

\PreparationTitle{Saute Pork}
\PreparationStep{Garlic}{Add}
\PreparationStep{Diced Pork}{Add}
\PreparationStep{Salt}{applied generously to meat}
\PreparationStep{Black pepper}{applied generously to meat}
\CookingStep{Fry at high heat}
\CookingCondition{Pork is golden outside}
\PreparationStep{Move to cooking pot}

\PreparationTitle{Prepare Bacon}
\IngredientPreparation{Chicken Stock}{Dissolve 1 stock cube in 250ml water}
\PreparationStep{Using skillet}
\IngredientPreparation{Smoked Bacon}{Add}
\IngredientPreparation{Paprika}{Add}
\IngredientPreparation{Chilli Powder}{Add}
\IngredientPreparation{Majoram}{Add}
\IngredientPreparation{Bayleaves}{Add}
\IngredientPreparation{Allspice}{Add}
\CookingStep{Fry at high heat}
\CookingCondition{Bacon starts rendering some fat, and turns golden}
\IngredientPreparation{Cider}{Add}
\CookingCondition{Bring to near-boil}
\PreparationStep{Chicken Stock}{Add}
\PreparationStep{Move to cooking pot}

\PreparationTitle{Cooking period One}
\CookingStep{Cover and cook in pot at medium-low heat}
\CookingCondition{90-minutes elapsed}

\PreparationTitle{Preparation of vegetables}
\IngredientPreparation{Carrot}{Peeled and sliced into rounds}

\PreparationTitle{Add vegetables}
\IngredientPreparation{Mushroom}{Sliced}
\IngredientPreparation{Tomato Concentrate}{Add}

\PreparationTitle{Cooking period Two}
\CookingStep{Cover and cook in pot at medium-low heat}
\CookingCondition{60-minutes elapsed, and meat is tender}

\PreparationTitle{Thicken broth}
\IngredientAmount{Cornflour}{Add to thicken}
\CookingStep{Stir, and bring to boil}

\EndCookingMethod
