% Cooking Instructions
\begin{enumerate}
\PreparationTitle{Par-boil Potatoes}
\IngredientPreparation{New Potatoes}{Cut into 1-inch chunks}
\PreparationStep{Place potatoes in a pot for par-boiling}
\PreparationStep{Add cold water, cover}
\CookingStep{Boil at high heat}
\CookingCondition{12-minutes elapsed from water boiled}

\PreparationTitle{Prepare tray Ingredients}
\PreparationStep{Warm fan-assisted oven to \ovenTemp{180} }
\IngredientPreparation{Shallots}{Peel, top, tail and roughly slice to 2cm blocks}
\IngredientPreparation{Chestnut Mushrooms}{Cut into quarters}
\IngredientPreparation{Apple}{Sliced into 2mm x 5mm x5mm slices}
\IngredientPreparation{Kiełbasa Krakowska}{Cut into 5mm half-moons}
\IngredientPreparation{Smoked Bacon}{Sliced into small pieces (max 1cm squares)}

\PreparationTitle{Prepare late Ingredients}
\PreparationStep{Mangetout}{Wash}
\PreparationStep{Baby Sweetcorn}{Wash and cut into 2cm long chunks}
\PreparationStep{Spring onions}{Peel and cut innto 5mm long chunks}

\PreparationTitle{Baking tray preparation}
\PreparationStep{Prepare large baking tray}
\PreparationStep{Shallots}{Spread on baking tray}
\PreparationStep{Chestnut Mushrooms}{Spread on baking tray}
\PreparationStep{Smoked Bacon}{Spread on baking tray}
\PreparationStep{Kiełbasa Krakowska}{Spread on baking tray}
\PreparationStep{Majoram}{Season}
\PreparationStep{Allspice}{Sprinkle}
\PreparationStep{Garlic Granules}{Sprinkle}
\PreparationStep{Smoked Pakrika}{Disperse onto chicken/potatoes}
\PreparationStep{Fresh thyme}{Sprinkle}
\PreparationStep{Salt}{Season}
\PreparationStep{Black pepper}{Sprinkle}
\CookingStep{Add boiled potatoes}
\PreparationStep{Olive Oil}{Spread over the ingredients}
\PreparationStep{Apple Cider}{Pour over the ingredients}

\PreparationTitle{Bake first round}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{20-minutes elapsed}

\PreparationTitle{Bake all ingredients}
\CookingStep{Ensure all ingredients rotated}
\PreparationStep{Mangetout}{Add}
\PreparationStep{Baby Sweetcorn}{Add}
\PreparationStep{Spring onions}{Add}
\CookingStep{Put into the pre-heated oven}
\CookingCondition{15-minutes elapsed}
